>>> import uqbar.apis
>>> documenter = uqbar.apis.ModuleDocumenter(
...     "uqbar.io",
...     module_documenters=[
...         uqbar.apis.ModuleDocumenter("uqbar.io.Timer"),
...     ],
... )
>>> print(str(documenter))
